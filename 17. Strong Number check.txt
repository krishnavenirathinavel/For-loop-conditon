
#include <stdio.h>

int main()
{
  int N, original, digit;
  int sum = 0;
  scanf("%d", &N);
  original = N;
  for ( ; N > 0; N = N / 10){
    digit = N % 10;

    int fact = 1;
    for (int i = 1; i <= digit; i++)
    {
      fact = fact * i;
      }
      sum = sum + fact;   
    }
    if (sum == original)
    printf("Yes");
    else
    printf("No");
    return 0;
}
